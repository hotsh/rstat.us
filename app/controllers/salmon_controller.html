class SalmonController < ApplicationController
  def feeds
    feed = Feed.first :id => params[:id]

    if feed.nil?
      render :file => "#{Rails.root}/public/404.html", :status => 404
      return
    end

    body = request.body.read
    if body.length == 0
      render :file => "#{Rails.root}/public/404.html", :status => 404
      return
    end

    salmon = OStatus::Salmon.from_xml body
    if salmon.nil?
      render :file => "#{Rails.root}/public/404.html", :status => 404
      return
    end
